S1medium DOUBLE,
S2medium DOUBLE,
S3medium DOUBLE,
P0medium DOUBLE,
P1medium DOUBLE,
P2medium DOUBLE,
P3medium DOUBLE
);

CREATE TABLE CITY(
name VARCHAR(30) PRIMARY KEY,
latitude DOUBLE,
longitude DOUBLE
);

CREATE TABLE USER(
id INT PRIMARY KEY NOT NULL AUTO_INCREMENT,
year INTEGER,
month INTEGER,
day INTEGER,
hour INTEGER,
minutes INTEGER,
seconds INTEGER,
follow BOOLEAN,
place VARCHAR(30) REFERENCES CITY(name),
planet VARCHAR(30) REFERENCES ASTRE(name)
);
